        TS TYPES


- String
    let var: string;

- Number
    let var: number;
- Boolean
    let var: boolean;

- Array
    let arr : Array<type>;
    let arr: type[];

- Any
    let var: any;

- Void
    let var: void;
    
- Null (void, null and undefined behave similarly)
    let var: null;

- Tuple
    let strNumTuple: [string,number];
    strNumTuple = ["Word", 10];
